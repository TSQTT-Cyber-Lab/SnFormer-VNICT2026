```python
from pathlib import Path
import shutil
import os

CODE_SRC = Path("/kaggle/input/snformer-code")
WORK = Path("/kaggle/working/SnFormer-VNICT2026")

if WORK.exists():
    shutil.rmtree(WORK)

shutil.copytree(
    CODE_SRC,
    WORK,
    ignore=shutil.ignore_patterns(
        ".git",
        ".venv",
        "__pycache__",
        "checkpoints",
        "results",
    ),
)

os.chdir(WORK)
print("cwd:", os.getcwd())
```

7. Cell 4 - Tao manifest CSV tu dataset tren Kaggle

Sua DATA_ROOT neu slug dataset cua ban khac snformer-data.

```python
from pathlib import Path
import csv
from collections import Counter

DATA_ROOT = Path("/kaggle/input/snformer-data")
OUT = Path("data/kaggle_manifest.csv")
OUT.parent.mkdir(exist_ok=True)

VIDEO_EXTS = {".mp4", ".avi", ".mov", ".mkv", ".webm"}

def label_for(path: Path):
    s = str(path).lower()
    if "celeb-real" in s or "youtube-real" in s:
        return 0
    if (
        "celeb-synthesis" in s
        or "deepfaketimit" in s
        or "deepfakes_v1" in s
        or "deepfakes_v4" in s
        or "higher_quality" in s
        or "lower_quality" in s
    ):
        return 1
    return None

rows = []
for p in DATA_ROOT.rglob("*"):
    if p.is_file() and p.suffix.lower() in VIDEO_EXTS:
        label = label_for(p)
        if label is not None:
            rows.append((str(p), label, ""))

rows.sort()

with OUT.open("w", newline="", encoding="utf-8") as f:
    w = csv.writer(f)
    w.writerow(["path", "label", "text"])
    w.writerows(rows)

print("manifest:", OUT)
print("total:", len(rows))
print("labels:", Counter(r[1] for r in rows))
```

12. Cell 9 - Tao file TXT tom tat tren Kaggle

Neu muon tao TXT tom tat nhanh:

```bash
!python - <<'PY'
import json
from pathlib import Path

history = json.loads(Path("results/training_history.json").read_text())
bench = json.loads(Path("results/benchmark_results.json").read_text())

lines = []
lines.append("SFORMER / SNFORMER KAGGLE RUN SUMMARY")
lines.append("")
lines.append("Training History")
for stage, rows in history.items():
    lines.append("")
    lines.append(stage)
    lines.append("epoch | loss | val_loss | val_acc")
    for r in rows:
        lines.append(f"{r['epoch']} | {r['loss']} | {r['val_loss']} | {r['val_acc']}")

lines.append("")
lines.append("Latency")
for name, row in bench.get("latency", {}).items():
    lines.append(
        f"{name}: {row['latency_ms']} ms/call, "
        f"{row['latency_per_frame']} ms/frame, "
        f"{row['fps']} FPS, params={row['params_M']}M"
    )

Path("results/kaggle_run_summary.txt").write_text("\n".join(lines), encoding="utf-8")
print("saved: results/kaggle_run_summary.txt")
PY
```
